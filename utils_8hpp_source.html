166  inline large_page_allocation calculate_large_page_allocation(size_t bytes)
167  {
168  large_page_allocation ret;
169  auto pagesizes(page_sizes());
170  do
171  {
172  ret.page_size_used = pagesizes.back();
173  pagesizes.pop_back();
174  } while(!pagesizes.empty() && !(bytes / ret.page_size_used));
175  ret.actual_size = (bytes + ret.page_size_used - 1) & ~(ret.page_size_used - 1);
176  return ret;
177  }
