152  // LOCK MUST BE HELD ON ENTRY!
153  void __post_done(post_info *pi)
154  {
155  // Find the post_info and remove it
156  for(auto &i : _posts)
157  {
158  if(&i == pi)
159  {
160  i.f.reset();
161  i.service = nullptr;
162  pi = nullptr;
163  break;
164  }
165  }
166  assert(!pi);
167  if(pi)
168  abort();
169  _work_done();
170  while(!_posts.front().service)
171  _posts.pop_front();
172  }

176  return __post_done(pi);
177  }
