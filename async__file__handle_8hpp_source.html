129  static inline result<async_file_handle> async_random_file(io_service &service, path_type dirpath, mode _mode = mode::write, caching _caching = caching::temporary, flag flags = flag::none) noexcept
130  {
131  try
132  {
133  result<async_file_handle> ret;
134  do
135  {
136  auto randomname = utils::random_string(32);
137  randomname.append(".random");
138  ret = async_file(service, dirpath / randomname, _mode, creation::only_if_not_exist, _caching, flags);
139  if(!ret && ret.get_error().value() != EEXIST)
140  return ret;

204  // Holds state for an i/o in progress. Will be subclassed with platform specific state and how to implement completion.
205  // Note this is allocated using malloc not new to avoid memory zeroing, and therefore it has a custom deleter.
206  struct _erased_io_state_type
207  {
208  async_file_handle *parent;
209  operation_t operation;
210  size_t items;
211  shared_size_type items_to_go;
212  constexpr _erased_io_state_type(async_file_handle *_parent, operation_t _operation, size_t _items)
213  : parent(_parent)
214  , operation(_operation)
215  , items(_items)
216  , items_to_go(0)
217  {
218  }
219  /*
220  For Windows:
221  - errcode: GetLastError() code
222  - bytes_transferred: obvious
223  - internal_state: LPOVERLAPPED for this op
224 
225  For POSIX AIO:
226  - errcode: errno code
227  - bytes_transferred: return from aio_return(), usually bytes transferred
228  - internal_state: address of pointer to struct aiocb in io_service's _aiocbsv
229  */
230  virtual void operator()(long errcode, long bytes_transferred, void *internal_state) noexcept = 0;
231  BOOST_AFIO_HEADERS_ONLY_VIRTUAL_SPEC ~_erased_io_state_type()
232  {
233  // i/o still pending is very bad, this should never happen
234  assert(!items_to_go);
235  if(items_to_go)
236  {
237  BOOST_AFIO_LOG_FATAL(parent->native_handle().h, "FATAL: io_state destructed while i/o still in flight, the derived class should never allow this.");
238  abort();
239  }
240  }
241  };
