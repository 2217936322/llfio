130  // Holds state for an i/o in progress. Will be subclassed with platform specific state and how to implement completion.
131  // Note this is allocated using malloc not new to avoid memory zeroing, and therefore it has a custom deleter.
132  struct _erased_io_state_type
133  {
134  async_file_handle *parent;
135  operation_t operation;
136  size_t items;
137  shared_size_type items_to_go;
138  constexpr _erased_io_state_type(async_file_handle *_parent, operation_t _operation, size_t _items)
139  : parent(_parent)
140  , operation(_operation)
141  , items(_items)
142  , items_to_go(0)
143  {
144  }
145  /*
146  For Windows:
147  - errcode: GetLastError() code
148  - bytes_transferred: obvious
149  - internal_state: LPOVERLAPPED for this op
150 
151  For POSIX AIO:
152  - errcode: errno code
153  - bytes_transferred: return from aio_return(), usually bytes transferred
154  - internal_state: address of pointer to struct aiocb in io_service's _aiocbsv
155  */
156  virtual void operator()(long errcode, long bytes_transferred, void *internal_state) noexcept = 0;
157  BOOST_AFIO_HEADERS_ONLY_VIRTUAL_SPEC ~_erased_io_state_type()
158  {
159  // i/o still pending is very bad, this should never happen
160  assert(!items_to_go);
161  if(items_to_go)
162  {
163  BOOST_AFIO_LOG_FATAL(parent->native_handle().h, "FATAL: io_state destructed while i/o still in flight, the derived class should never allow this.");
164  abort();
165  }
166  }
167  };
168  // State for an i/o in progress, but with the per operation typing
