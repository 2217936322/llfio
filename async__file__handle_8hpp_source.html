128  // Holds state for an i/o in progress. Will be subclassed with platform specific state and how to implement completion.
129  // Note this is allocated using malloc not new to avoid memory zeroing, and therefore it has a custom deleter.
130  struct _erased_io_state_type
131  {
132  async_file_handle *parent;
133  operation_t operation;
134  size_t items;
135  shared_size_type items_to_go;
136  constexpr _erased_io_state_type(async_file_handle *_parent, operation_t _operation, size_t _items)
137  : parent(_parent)
138  , operation(_operation)
139  , items(_items)
140  , items_to_go(0)
141  {
142  }
143  /*
144  For Windows:
145  - errcode: GetLastError() code
146  - bytes_transferred: obvious
147  - internal_state: LPOVERLAPPED for this op
148 
149  For POSIX AIO:
150  - errcode: errno code
151  - bytes_transferred: return from aio_return(), usually bytes transferred
152  - internal_state: address of pointer to struct aiocb in io_service's _aiocbsv
153  */
154  virtual void operator()(long errcode, long bytes_transferred, void *internal_state) noexcept = 0;
155  BOOST_AFIO_HEADERS_ONLY_VIRTUAL_SPEC ~_erased_io_state_type()
156  {
157  // i/o still pending is very bad, this should never happen
158  assert(!items_to_go);
159  if(items_to_go)
160  {
161  BOOST_AFIO_LOG_FATAL(parent->native_handle().h, "FATAL: io_state destructed while i/o still in flight, the derived class should never allow this.");
162  abort();
163  }
164  }
165  };
166  // State for an i/o in progress, but with the per operation typing

172  : _erased_io_state_type(_parent, _operation, _items)
173  , result(make_result(BuffersType()))
174  , completion(std::forward<CompletionRoutine>(f))

176  }
177  };
