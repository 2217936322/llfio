#define BOOST_AFIO_HEADERS_ONLY   1
 Whether AFIO is a headers only library. Defaults to 1 unless BOOST_ALL_DYN_LINK is defined.
 
#define BOOST_AFIO_LOGGING_LEVEL   4
 How much detail to log. 0=disabled, 1=fatal, 2=error, 3=warn, 4=info, 5=debug, 6=all. Defaults to error if NDEBUG defined, else info level.
 
#define BOOST_AFIO_LOG_BACKTRACE_LEVELS   ((1 << 1) | (1 << 2) | (1 << 3))
 Bit mask of which log levels should be stack backtraced which will slow those logs thirty fold or so. Defaults to (1<<1)|(1<<2)|(1<<3) i.e. stack backtrace on fatal, error and warn logs.
 
#define BOOST_AFIO_LOGGING_MEMORY   (1024 * 1024)
 How much memory to use for the log. Defaults to 4Kb if NDEBUG defined, else 1Mb.
 
#define BOOST_AFIO_V2_STL11_IMPL   std

#define BOOST_AFIO_USE_BOOST_THREAD   0
 Whether to use Boost.Thread instead of the C++ 11 STL std::thread. Defaults to the C++ 11 STL thread.
 
#define BOOST_AFIO_USE_BOOST_FILESYSTEM   1
 Whether to use Boost.Filesystem instead of the C++ 17 TS std::filesystem. Defaults to the C++ 17 TS filesystem if that is available, else Boost.
