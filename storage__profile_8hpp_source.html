95  template <class T> struct item : public item_base
96  {
97  static constexpr size_t item_size = item_base::item_size;
98  using handle_type = item_base::handle_type;
99  using callable = outcome<void> (*)(storage_profile &sp, handle_type &h);
100 
101  callable impl;
102  T value;
103  char _padding[item_size - sizeof(item_base) - sizeof(callable) - sizeof(T)];
104  constexpr item(const char *_name, callable c, const char *_desc = nullptr, T _value = default_value<T>())
105  : item_base(_name, _desc, map_to_storage_type<T>())
106  , impl(c)
107  , value(_value)
108  , _padding{0}
109  {
110  static_assert(sizeof(*this) == item_size, "");
111  }
113  void clear() { value = default_value<T>(); }
115  outcome<void> operator()(storage_profile &sp, handle_type &h) const
116  {
117  if(value != default_value<T>())
118  return make_ready_outcome<void>();
119  return impl(sp, h);
120  }

136  template <class U> auto invoke(U &&f) const
137  {
138  switch(type)
139  {
140  case storage_types::extent_type:
141  return f(*static_cast<const item<io_service::extent_type> *>(static_cast<const item_base *>(this)));
142  case storage_types::unsigned_int:
143  return f(*static_cast<const item<unsigned int> *>(static_cast<const item_base *>(this)));
144  case storage_types::unsigned_long_long:
145  return f(*static_cast<const item<unsigned long long> *>(static_cast<const item_base *>(this)));
146  case storage_types::float_:
147  return f(*static_cast<const item<float> *>(static_cast<const item_base *>(this)));
148  case storage_types::string:
149  return f(*static_cast<const item<std::string> *>(static_cast<const item_base *>(this)));
150  case storage_types::unknown:
151  break;
152  }
153  throw std::invalid_argument("No type set in item");
154  }
156  outcome<void> operator()(storage_profile &sp, handle_type &h) const
157  {
158  return invoke([&sp, &h](auto &item) { return item(sp, h); });
159  }
