93  template <class T> struct item : public item_base
94  {
95  static constexpr size_t item_size = item_base::item_size;
96  using handle_type = item_base::handle_type;
97  using callable = outcome<void> (*)(storage_profile &sp, handle_type &h);
98 
99  callable impl;
100  T value;
101  char _padding[item_size - sizeof(item_base) - sizeof(callable) - sizeof(T)];
102  constexpr item(const char *_name, callable c, const char *_desc = nullptr, T _value = default_value<T>())
103  : item_base(_name, _desc, map_to_storage_type<T>())
104  , impl(c)
105  , value(_value)
106  , _padding{0}
107  {
108  static_assert(sizeof(*this) == item_size, "");
109  }
111  void clear() { value = default_value<T>(); }
113  outcome<void> operator()(storage_profile &sp, handle_type &h) const
114  {
115  if(value != default_value<T>())
116  return make_ready_outcome<void>();
117  return impl(sp, h);
118  }
119  };
121  struct item_erased : public item_base
122  {
123  static constexpr size_t item_size = item_base::item_size;
124  using handle_type = item_base::handle_type;
125  char _padding[item_size - sizeof(item_base)];
126 
127  item_erased() = delete;
128  ~item_erased() = delete;
129  item_erased(const item_erased &) = delete;
130  item_erased(item_erased &&) = delete;
131  item_erased &operator=(const item_erased &) = delete;
132  item_erased &operator=(item_erased &&) = delete;
134  template <class U> auto invoke(U &&f) const
135  {
136  switch(type)
137  {
138  case storage_types::extent_type:
139  return f(*static_cast<const item<io_service::extent_type> *>(static_cast<const item_base *>(this)));
140  case storage_types::unsigned_int:
141  return f(*static_cast<const item<unsigned int> *>(static_cast<const item_base *>(this)));
142  case storage_types::unsigned_long_long:
143  return f(*static_cast<const item<unsigned long long> *>(static_cast<const item_base *>(this)));
144  case storage_types::float_:
145  return f(*static_cast<const item<float> *>(static_cast<const item_base *>(this)));
146  case storage_types::string:
147  return f(*static_cast<const item<std::string> *>(static_cast<const item_base *>(this)));
148  case storage_types::unknown:
149  break;
150  }
151  throw std::invalid_argument("No type set in item");
152  }
154  outcome<void> operator()(storage_profile &sp, handle_type &h) const
155  {
156  return invoke([&sp, &h](auto &item) { return item(sp, h); });
157  }
158  };
