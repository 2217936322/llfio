69  }
71  template <class Rep, class Period>
72  deadline(stl11::chrono::duration<Rep, Period> d)
73  : steady(true)
74  {
75  stl11::chrono::nanoseconds _nsecs = stl11::chrono::duration_cast<stl11::chrono::nanoseconds>(d);
76  // Negative durations are zero duration
77  if(_nsecs.count() > 0)
78  nsecs = _nsecs.count();
79  else
80  nsecs = 0;
81  }
83  stl11::chrono::system_clock::time_point to_time_point() const
84  {
85  if(steady)
86  throw std::invalid_argument("Not a UTC deadline!");
87  stl11::chrono::system_clock::time_point tp(stl11::chrono::system_clock::from_time_t(utc.tv_sec));
88  tp += stl11::chrono::duration_cast<stl11::chrono::system_clock::duration>(stl11::chrono::nanoseconds(utc.tv_nsec));
89  return tp;
