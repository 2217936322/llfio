108  virtual result<void> _lock(entities_guard &out, deadline d, bool spin_not_sleep) noexcept override final
109  {
110  BOOST_AFIO_LOG_FUNCTION_CALL(this);
111  stl11::chrono::steady_clock::time_point began_steady;
112  stl11::chrono::system_clock::time_point end_utc;
113  if(d)
114  {
115  if((d).steady)
116  began_steady = stl11::chrono::steady_clock::now();
117  else
118  end_utc = (d).to_time_point();
119  }
120  // Fire this if an error occurs
121  auto disableunlock = detail::Undoer([&] { out.release(); });
122  size_t n;
123  for(;;)
124  {
125  size_t was_contended = (size_t) -1;
126  {
127  auto undo = detail::Undoer([&] {
128  // 0 to (n-1) need to be closed
129  if(n > 0)
130  {
131  --n;
132  // Now 0 to n needs to be closed
133  for(; n > 0; n--)
134  _h.unlock(out.entities[n].value, 1);
135  _h.unlock(out.entities[0].value, 1);
136  }
137  });
138  for(n = 0; n < out.entities.size(); n++)
139  {
140  deadline nd;
141  // Only for very first entity will we sleep until its lock becomes available
142  if(n)
143  nd = deadline(stl11::chrono::seconds(0));
144  else
145  {
146  nd = deadline();
147  if(d)
148  {
149  if((d).steady)
150  {
151  stl11::chrono::nanoseconds ns = stl11::chrono::duration_cast<stl11::chrono::nanoseconds>((began_steady + stl11::chrono::nanoseconds((d).nsecs)) - stl11::chrono::steady_clock::now());
152  if(ns.count() < 0)
153  (nd).nsecs = 0;
154  else
155  (nd).nsecs = ns.count();
156  }
157  else
158  (nd) = (d);
159  }
160  }
161  auto outcome = _h.lock(out.entities[n].value, 1, out.entities[n].exclusive, nd);
162  if(!outcome)
163  {
164  was_contended = n;
165  goto failed;
166  }
167  outcome.get().release();
168  }
169  // Everything is locked, exit
170  undo.dismiss();
171  disableunlock.dismiss();
172  return make_result<void>();
173  }
174  failed:
175  if(d)
176  {
177  if((d).steady)
178  {
179  if(stl11::chrono::steady_clock::now() >= (began_steady + stl11::chrono::nanoseconds((d).nsecs)))
180  return make_errored_result<void>(ETIMEDOUT);
181  }
182  else
183  {
184  if(stl11::chrono::system_clock::now() >= end_utc)
185  return make_errored_result<void>(ETIMEDOUT);
186  }
187  }
188 #if 1
189  // Move was_contended to front and randomise rest of out.entities
190  std::swap(out.entities[was_contended], out.entities[0]);
191  auto front = out.entities.begin();
192  ++front;
193  std::random_shuffle(front, out.entities.end());
194 #else
195  std::random_shuffle(out.entities.begin(), out.entities.end());
196 #endif
197  if(!spin_not_sleep)
198  std::this_thread::yield();
199  }
200  // return make_result<void>();
201  }
