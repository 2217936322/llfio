546  }
547  *bufferp = 0;

553 {

555  const span<char> remove2 = outcome_namespace_string();

560  if(!memcmp(in, remove2.data(), remove2.size()))
561  in += remove2.size();
562  *out++ = *in++;
563  }
564  *out = 0;
