628  {
629  bool _dismissed;
630  callable undoer;
631  UndoerImpl() = delete;
632  UndoerImpl(const UndoerImpl &) = delete;
633  UndoerImpl &operator=(const UndoerImpl &) = delete;
634  explicit UndoerImpl(callable &&c)
635  : _dismissed(false)
636  , undoer(std::move(c))
637  {
638  }
639  void int_trigger()
640  {
641  if(!_dismissed && !is_nullptr(undoer))
642  {
643  undoer();
644  _dismissed = true;
645  }

651  {

655  o._dismissed = true;
656  return *this;
657  }
