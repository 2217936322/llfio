50  {

130  result<void> _read_header()
131  {
132  bool first = true;
133  do
134  {
135  BOOST_OUTCOME_FILTER_ERROR(_, _h.read(0, (char *) &_header, 48));
136  if(_.first != (char *) &_header)
137  memcpy(&_header, _.first, _.second);
138  if(_skip_hashing)
139  return make_result<void>();
140  if(first)
141  first = false;
142  else
143  stl11::this_thread::yield();
144  // No timeout as this should very rarely block for any significant length of time
145  } while(_header.hash != utils::fast_hash::hash(((char *) &_header) + 16, sizeof(_header) - 16));
146  return make_result<void>();
147  }

175  static result<atomic_append> fs_mutex_append(file_handle::path_type lockfile, bool nfs_compatibility = false, bool skip_hashing = false) noexcept
176  {
177  BOOST_AFIO_LOG_FUNCTION_CALL(0);
178  BOOST_OUTCOME_FILTER_ERROR(ret, file_handle::file(std::move(lockfile), file_handle::mode::write, file_handle::creation::if_needed, file_handle::caching::temporary, file_handle::flag::delete_on_close));
179  atomic_append_detail::header header;
180  // Lock the entire header for exclusive access
181  auto lockresult = ret.try_lock(0, sizeof(header), true);
182  if(lockresult.has_error())
183  {
184  if(lockresult.get_error().value() != ETIMEDOUT)
185  return lockresult.get_error();
186  // Somebody else is also using this file
187  }
188  else
189  {
190  // I am the first person to be using this (stale?) file, so write a new header and truncate
191  ret.truncate(sizeof(header));
192  memset(&header, 0, sizeof(header));
193  header.time_offset = stl11::chrono::system_clock::to_time_t(stl11::chrono::system_clock::now());
194  header.first_known_good = sizeof(header);
195  header.first_after_hole_punch = sizeof(header);
196  if(!skip_hashing)
197  header.hash = utils::fast_hash::hash(((char *) &header) + 16, sizeof(header) - 16);
198  BOOST_OUTCOME_FILTER_ERROR(_, ret.write(0, (char *) &header, sizeof(header)));
199  (void) _;
200  }
201  // Open a shared lock on last byte in header to prevent other users zomping the file
202  BOOST_OUTCOME_FILTER_ERROR(guard, ret.lock(sizeof(header) - 1, 1, false));
203  // Unlock any exclusive lock I gained earlier now
204  if(lockresult)
205  lockresult.get().unlock();
206  // The constructor will read and cache the header
207  return atomic_append(std::move(ret), std::move(guard), nfs_compatibility, skip_hashing);
208  }

214  virtual result<void> _lock(entities_guard &out, deadline d, bool spin_not_sleep) noexcept override final
215  {
216  BOOST_AFIO_LOG_FUNCTION_CALL(this);
217  atomic_append_detail::lock_request lock_request;
218  if(out.entities.size() > sizeof(lock_request.entities) / sizeof(lock_request.entities[0]))
219  return make_errored_result<void>(E2BIG);
220 
221  stl11::chrono::steady_clock::time_point began_steady;
222  stl11::chrono::system_clock::time_point end_utc;
223  if(d)
224  {
225  if((d).steady)
226  began_steady = stl11::chrono::steady_clock::now();
227  else
228  end_utc = (d).to_time_point();
229  }
230  // Fire this if an error occurs
231  auto disableunlock = detail::Undoer([&] { out.release(); });
232 
233  // Write my lock request immediately
234  memset(&lock_request, 0, sizeof(lock_request));
235  lock_request.unique_id = _unique_id;
236  auto count = stl11::chrono::system_clock::now() - stl11::chrono::system_clock::from_time_t(_header.time_offset);
237  lock_request.us_count = stl11::chrono::duration_cast<stl11::chrono::microseconds>(count).count();
238  lock_request.items = out.entities.size();
239  memcpy(lock_request.entities, out.entities.data(), sizeof(lock_request.entities[0]) * out.entities.size());
240  if(!_skip_hashing)
241  lock_request.hash = utils::fast_hash::hash(((char *) &lock_request) + 16, sizeof(lock_request) - 16);
242  // My lock request will be the file's current length or higher
243  BOOST_OUTCOME_FILTER_ERROR(my_lock_request_offset, _h.length());
244  {
245  _h.set_append_only(true);
246  auto undo = detail::Undoer([this] { _h.set_append_only(false); });
247  file_handle::extent_guard append_guard;
248  if(_nfs_compatibility)
249  {
250  file_handle::extent_type lastbyte = (file_handle::extent_type) -1;
251  // Lock up to the beginning of the shadow lock space
252  lastbyte &= ~(1ULL << 63);
253  BOOST_OUTCOME_FILTER_ERROR(append_guard_, _h.lock(my_lock_request_offset, lastbyte, true));
254  append_guard = std::move(append_guard_);
255  }
256  BOOST_OUTCOME_FILTER_ERROR(_, _h.write(0, (char *) &lock_request, sizeof(lock_request)));
257  (void) _;
258  }
259 
260  // Find the record I just wrote
261  alignas(64) char _buffer[4096 + 2048]; // 6Kb cache line aligned buffer
262  // Read onwards from length as reported before I wrote my lock request
263  // until I find my lock request. This loop should never actually iterate
264  // except under extreme load conditions.
265  for(;;)
266  {
267  file_handle::io_result<file_handle::buffer_type> readoutcome = _h.read(my_lock_request_offset, _buffer, sizeof(_buffer));
268  // Should never happen :)
269  if(readoutcome.has_error())
270  {
271  BOOST_AFIO_LOG_FATAL(this, "atomic_append::lock() saw an error when searching for just written data");
272  std::terminate();
273  }
274  const atomic_append_detail::lock_request *record, *lastrecord;
275  for(record = (const atomic_append_detail::lock_request *) readoutcome.get().first, lastrecord = (const atomic_append_detail::lock_request *) (readoutcome.get().first + readoutcome.get().second); record < lastrecord && record->hash != lock_request.hash; ++record)
276  my_lock_request_offset += sizeof(atomic_append_detail::lock_request);
277  if(record->hash == lock_request.hash)
278  break;
279  }
280 
281  // extent_guard is now valid and will be unlocked on error
282  out.hint = my_lock_request_offset;
283  disableunlock.dismiss();
284 
285  // Lock my request for writing so others can sleep on me
286  file_handle::extent_guard my_request_guard;
287  if(!spin_not_sleep)
288  {
289  auto lock_offset = my_lock_request_offset;
290  // Set the top bit to use the shadow lock space on Windows
291  lock_offset |= (1ULL << 63);
292  BOOST_OUTCOME_FILTER_ERROR(my_request_guard_, _h.lock(lock_offset, sizeof(lock_request), true));
293  my_request_guard = std::move(my_request_guard_);
294  }
295 
296  // Read every record preceding mine until header.first_known_good inclusive
297  auto record_offset = my_lock_request_offset - sizeof(atomic_append_detail::lock_request);
298  do
299  {
300  reload:
301  // Refresh the header and load a snapshot of everything between record_offset
302  // and first_known_good or -6Kb, whichever the sooner
303  _read_header();
304  // If there are no preceding records, we're done
305  if(record_offset < _header.first_known_good)
306  break;
307  auto start_offset = record_offset;
308  if(start_offset > sizeof(_buffer) - sizeof(atomic_append_detail::lock_request))
309  start_offset -= sizeof(_buffer) - sizeof(atomic_append_detail::lock_request);
310  else
311  start_offset = sizeof(atomic_append_detail::lock_request);
312  if(start_offset < _header.first_known_good)
313  start_offset = _header.first_known_good;
314  assert(record_offset >= start_offset);
315  assert(record_offset - start_offset <= sizeof(_buffer));
316  BOOST_OUTCOME_FILTER_ERROR(batchread, _h.read(start_offset, _buffer, record_offset - start_offset + sizeof(atomic_append_detail::lock_request)));
317  assert(batchread.second == record_offset - start_offset + sizeof(atomic_append_detail::lock_request));
318  const atomic_append_detail::lock_request *record = (atomic_append_detail::lock_request *) (batchread.first + batchread.second - sizeof(atomic_append_detail::lock_request));
319  const atomic_append_detail::lock_request *firstrecord = (atomic_append_detail::lock_request *) batchread.first;
320 
321  // Skip all completed lock requests or not mentioning any of my entities
322  for(; record >= firstrecord; record_offset -= sizeof(atomic_append_detail::lock_request), --record)
323  {
324  // If a completed lock request, skip
325  if(!record->hash && !record->unique_id)
326  continue;
327  // If record hash doesn't match contents it's a torn read, reload
328  if(!_skip_hashing)
329  {
330  if(record->hash != utils::fast_hash::hash(((char *) record) + 16, sizeof(atomic_append_detail::lock_request) - 16))
331  goto reload;
332  }
333 
334  // Does this record lock anything I am locking?
335  for(const auto &entity : out.entities)
336  {
337  for(size_t n = 0; n < record->items; n++)
338  {
339  if(record->entities[n].value == entity.value)
340  {
341  // Is the lock I want exclusive or the lock he wants exclusive?
342  // If so, need to block
343  if(record->entities[n].exclusive || entity.exclusive)
344  goto beginwait;
345  }
346  }
347  }
348  }
349  // None of this batch of records has anything to do with my request, so keep going
350  continue;
351 
352  beginwait:
353  // Sleep until this record is freed using a shared lock
354  // on the record in our way. Note there is a race here
355  // between when the lock requester writes the lock
356  // request and when he takes an exclusive lock on it,
357  // so if our shared lock succeeds we need to immediately
358  // unlock and retry based on the data.
359  stl11::this_thread::yield();
360  if(!spin_not_sleep)
361  {
362  deadline nd;
363  if(d)
364  {
365  if((d).steady)
366  {
367  stl11::chrono::nanoseconds ns = stl11::chrono::duration_cast<stl11::chrono::nanoseconds>((began_steady + stl11::chrono::nanoseconds((d).nsecs)) - stl11::chrono::steady_clock::now());
368  if(ns.count() < 0)
369  (nd).nsecs = 0;
370  else
371  (nd).nsecs = ns.count();
372  }
373  else
374  (nd) = (d);
375  }
376  auto lock_offset = record_offset;
377  // Set the top bit to use the shadow lock space on Windows
378  lock_offset |= (1ULL << 63);
379  BOOST_OUTCOME_FILTER_ERROR(record_guard_, _h.lock(lock_offset, sizeof(record), false, nd));
380  }
381  // Make sure we haven't timed out during this wait
382  if(d)
383  {
384  if((d).steady)
385  {
386  if(stl11::chrono::steady_clock::now() >= (began_steady + stl11::chrono::nanoseconds((d).nsecs)))
387  return make_errored_result<void>(ETIMEDOUT);
388  }
389  else
390  {
391  if(stl11::chrono::system_clock::now() >= end_utc)
392  return make_errored_result<void>(ETIMEDOUT);
393  }
394  }
395  } while(record_offset >= _header.first_known_good);
396  return make_result<void>();
397  }

400  virtual void unlock(entities_type entities, unsigned long long hint) noexcept override final
401  {
402  (void) entities;
403  BOOST_AFIO_LOG_FUNCTION_CALL(this);
404  if(!hint)
405  {
406  BOOST_AFIO_LOG_WARN(this, "atomic_append::unlock() currently requires a hint to work, assuming this is a failed lock.");
407  return;
408  }
409  file_handle::extent_type my_lock_request_offset = (file_handle::extent_type) hint;
410  {
411  atomic_append_detail::lock_request record;
412 #ifdef _DEBUG
413  (void) _h.read(my_lock_request_offset, (char *) &record, sizeof(record));
414  if(!record.unique_id)
415  {
416  BOOST_AFIO_LOG_FATAL(this, "atomic_append::unlock() I have been previously unlocked!");
417  std::terminate();
418  }
419  _read_header();
420  if(_header.first_known_good > my_lock_request_offset)
421  {
422  BOOST_AFIO_LOG_FATAL(this, "atomic_append::unlock() header exceeds the lock I am unlocking!");
423  std::terminate();
424  }
425 #endif
426  memset(&record, 0, sizeof(record));
427  (void) _h.write(my_lock_request_offset, (char *) &record, sizeof(record));
428  }
429 
430  // Every 32 records or so, bump _header.first_known_good
431  if(!(my_lock_request_offset & 4095))
432  {
433  //_read_header();
434 
435  // Forward scan records until first non-zero record is found
436  // and update header with new info
437  alignas(64) char _buffer[4096 + 2048];
438  bool done = false;
439  while(!done)
440  {
441  auto bytesread_ = _h.read(_header.first_known_good, _buffer, sizeof(_buffer));
442  if(bytesread_.has_error())
443  {
444  // If distance between original first known good and end of file is exactly
445  // 6Kb we can read an EOF
446  break;
447  }
448  const auto &bytesread = bytesread_.get();
449  // If read was partial, we are done after this round
450  if(bytesread.second < sizeof(_buffer))
451  done = true;
452  const atomic_append_detail::lock_request *record = (const atomic_append_detail::lock_request *) bytesread.first;
453  const atomic_append_detail::lock_request *lastrecord = (const atomic_append_detail::lock_request *) (bytesread.first + bytesread.second);
454  for(; record < lastrecord; ++record)
455  {
456  if(!record->hash && !record->unique_id)
457  _header.first_known_good += sizeof(atomic_append_detail::lock_request);
458  else
459  break;
460  }
461  }
462  // Hole punch if >= 1Mb of zeros exists
463  if(_header.first_known_good - _header.first_after_hole_punch >= 1024 * 1024)
464  {
465  off_t holepunchend = _header.first_known_good & ~(1024 * 1024 - 1);
466 #ifdef _DEBUG
467  fprintf(stderr, "hole_punch(%llx, %llx)\n", _header.first_after_hole_punch, holepunchend - _header.first_after_hole_punch);
468 #endif
469  _header.first_after_hole_punch = holepunchend;
470  }
471  ++_header.generation;
472  if(!_skip_hashing)
473  _header.hash = utils::fast_hash::hash(((char *) &_header) + 16, sizeof(_header) - 16);
474  // Rewrite the first part of the header only
475  (void) _h.write(0, (char *) &_header, 48);
476  }
477  }
