175  ~memory_map()
176  {
177  // Release my shared locks and try locking inuse exclusively
178  _hmapinuse.unlock();
179  _hlockinuse.unlock();
180  auto lockresult = _h.try_lock(_lockinuseoffset, 1, true);
181  if(lockresult)
182  {
183  // This means I am the last user, so zop the file contents as temp file is about to go away
184  char buffer[4096];

203  static result<memory_map> fs_mutex_map(file_handle::path_type lockfile, shared_fs_mutex *fallbacklock = nullptr) noexcept
204  {
205  BOOST_AFIO_LOG_FUNCTION_CALL(0);
206  try
207  {
208  BOOST_OUTCOME_FILTER_ERROR(ret, file_handle::file(std::move(lockfile), file_handle::mode::write, file_handle::creation::if_needed, file_handle::caching::reads));
209  file_handle temph;
210  // Am I the first person to this file? Lock the inuse exclusively
211  auto lockinuse = ret.try_lock(_lockinuseoffset, 1, true);
212  file_handle::extent_guard mapinuse;
213  if(lockinuse.has_error())
214  {
215  if(lockinuse.get_error().value() != ETIMEDOUT)
216  return lockinuse.get_error();
217  // Somebody else is also using this file, so try to read the hash index file I ought to use
218  lockinuse = ret.lock(_lockinuseoffset, 1, false); // last byte shared access
219  char buffer[65536];
220  memset(buffer, 0, sizeof(buffer));
221  {
222  BOOST_OUTCOME_FILTER_ERROR(_, ret.read(0, buffer, 65535));
223  (void) _;
224  }
225  fixme_path::value_type *temphpath = (fixme_path::value_type *) buffer;
226  result<file_handle> _temph;
227  // If path is zeroed, fall back onto backup lock
228  if(!buffer[0])
229  goto use_fall_back_lock;
230  else
231  _temph = file_handle::file(temphpath, file_handle::mode::write, file_handle::creation::open_existing, file_handle::caching::temporary);
232  // If temp file doesn't exist, I am on a different machine
233  if(!_temph)
234  {
235  // Zop the path so any new entrants into this lock will go to the fallback lock
236  memset(buffer, 0, 4096);
237  (void) ret.write(0, buffer, 4096);
238  use_fall_back_lock:
239  // I am guaranteed that all mmap users have locked the second last byte
240  // and will unlock it once everyone has stopped using the mmap, so make
241  // absolutely sure the mmap is not in use by anyone by taking an exclusive
242  // lock on the second final byte
243  BOOST_OUTCOME_FILTER_ERROR(mapinuse2, ret.lock(_mapinuseoffset, 1, true));
244  // Release the exclusive lock and tell caller to just use the fallback lock directly
245  return make_errored_result<memory_map>(EBUSY);
246  }
247  else
248  {
249  // Mark the map as being in use by me too
250  BOOST_OUTCOME_FILTER_ERROR(mapinuse2, ret.lock(_mapinuseoffset, 1, false));
251  mapinuse = std::move(mapinuse2);
252  temph = std::move(_temph.get());
253  }
254  }
255  else
256  {
257  // I am the first person to be using this (stale?) file, so create a new hash index file and write its path
258  ret.truncate(0);
259  BOOST_OUTCOME_FILTER_ERROR(_temph, file_handle::random_file(fixme_temporary_files_directory()));
260  temph = std::move(_temph);
261  auto temppath(temph.path());
262  temph.truncate(HashIndexSize);
263  // Write the path of my new hash index file and convert my lock to a shared one
264  {
265  BOOST_OUTCOME_FILTER_ERROR(_, ret.write(0, (const char *) temppath.c_str(), temppath.native().size() * sizeof(*temppath.c_str())));
266  (void) _;
267  }
268  // Convert exclusive whole file lock into lock in use
269  BOOST_OUTCOME_FILTER_ERROR(mapinuse2, ret.lock(_mapinuseoffset, 1, false));
270  BOOST_OUTCOME_FILTER_ERROR(lockinuse2, ret.lock(_lockinuseoffset, 1, false));
271  mapinuse = std::move(mapinuse2);
272  lockinuse = std::move(lockinuse2);
273  }
274  // Map the files into memory, being very careful that the lock file is only ever mapped read only
275  // as some OSs can get confused if you use non-mmaped writes on a region mapped for writing.
276  BOOST_OUTCOME_FILTER_ERROR(hsection, section_handle::section(ret, 0, section_handle::flag::read));
277  BOOST_OUTCOME_FILTER_ERROR(temphsection, section_handle::section(temph, HashIndexSize));
278  BOOST_OUTCOME_FILTER_ERROR(hmap, map_handle::map(hsection, 0, 0, section_handle::flag::read));
279  BOOST_OUTCOME_FILTER_ERROR(temphmap, map_handle::map(temphsection, HashIndexSize));
280  return memory_map(std::move(ret), std::move(temph), std::move(lockinuse.get()), std::move(mapinuse), std::move(hmap), std::move(temphmap), fallbacklock);
281  }
282  BOOST_OUTCOME_CATCH_EXCEPTION_TO_RESULT(memory_map)

294  // Create a cache of entities to their indices, eliding collisions where necessary
295  static span<_entity_idx> _hash_entities(_entity_idx *entity_to_idx, entities_type &entities)
296  {
297  _entity_idx *ep = entity_to_idx;
298  for(size_t n = 0; n < entities.size(); n++)
299  {
300  ep->value = hasher_type()(entities[n].value) % _container_entries;
301  ep->exclusive = entities[n].exclusive;
302  bool skip = false;
303  for(size_t m = 0; m < n; m++)
304  {
305  if(entity_to_idx[m].value == ep->value)
306  {
307  if(ep->exclusive && !entity_to_idx[m].exclusive)
308  entity_to_idx[m].exclusive = true;
309  skip = true;
310  }
311  }
312  if(!skip)
313  ++ep;
314  }
315  return span<_entity_idx>(entity_to_idx, ep - entity_to_idx);
316  }
317  virtual result<void> _lock(entities_guard &out, deadline d, bool spin_not_sleep) noexcept override final
318  {
319  BOOST_AFIO_LOG_FUNCTION_CALL(this);
320  if(is_degraded())
321  {
322  if(!_have_degraded)
323  {
324  // We have just become degraded, so release our shared lock of the map
325  // being in use and lock it exclusively, this will gate us on all other
326  // current users of the map so we don't unblock until all current users
327  // reach this same point. If that lock times out, we will reenter here
328  // next time until we succeed
329  _hmapinuse.unlock();
330  BOOST_OUTCOME_FILTER_ERROR(mapinuse2, _h.lock(_mapinuseoffset, 1, true, d));
331  _have_degraded = true;
332  }
333  if(_fallbacklock)
334  return _fallbacklock->_lock(out, d, spin_not_sleep);
335  return make_errored_result<void>(EBUSY);
336  }
337  stl11::chrono::steady_clock::time_point began_steady;
338  stl11::chrono::system_clock::time_point end_utc;
339  if(d)
340  {
341  if((d).steady)
342  began_steady = stl11::chrono::steady_clock::now();
343  else
344  end_utc = (d).to_time_point();
345  }
346  // alloca() always returns 16 byte aligned addresses
347  span<_entity_idx> entity_to_idx(_hash_entities((_entity_idx *) alloca(sizeof(_entity_idx) * out.entities.size()), out.entities));
348  _hash_index_type &index = _index();
349  // Fire this if an error occurs
350  auto disableunlock = undoer([&] { out.release(); });
351  size_t n;
352  for(;;)
353  {
354  size_t was_contended = (size_t) -1;
355  {
356  auto undo = undoer([&] {
357  // 0 to (n-1) need to be closed
358  if(n > 0)
359  {
360  --n;
361  // Now 0 to n needs to be closed
362  for(; n > 0; n--)
363  entity_to_idx[n].exclusive ? index[entity_to_idx[n].value].unlock() : index[entity_to_idx[n].value].unlock_shared();
364  entity_to_idx[0].exclusive ? index[entity_to_idx[0].value].unlock() : index[entity_to_idx[0].value].unlock_shared();
365  }
366  });
367  for(n = 0; n < entity_to_idx.size(); n++)
368  {
369  if(!(entity_to_idx[n].exclusive ? index[entity_to_idx[n].value].try_lock() : index[entity_to_idx[n].value].try_lock_shared()))
370  {
371  was_contended = n;
372  goto failed;
373  }
374  }
375  // Everything is locked, exit
376  undo.dismiss();
377  disableunlock.dismiss();
378  return make_result<void>();
379  }
380  failed:
381  if(d)
382  {
383  if((d).steady)
384  {
385  if(stl11::chrono::steady_clock::now() >= (began_steady + stl11::chrono::nanoseconds((d).nsecs)))
386  return make_errored_result<void>(ETIMEDOUT);
387  }
388  else
389  {
390  if(stl11::chrono::system_clock::now() >= end_utc)
391  return make_errored_result<void>(ETIMEDOUT);
392  }
393  }
394  // Move was_contended to front and randomise rest of out.entities
395  std::swap(entity_to_idx[was_contended], entity_to_idx[0]);
396  auto front = entity_to_idx.begin();
397  ++front;
398  std::random_shuffle(front, entity_to_idx.end());
399  if(!spin_not_sleep)
400  std::this_thread::yield();
401  }
402  // return make_result<void>();
403  }
